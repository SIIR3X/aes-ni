170static inline void aes256_key_assist(__m128i* temp1, __m128i* temp2, __m128i* temp3)
171{
172 __m128i temp4;
173
174 // Expand temp1 (first half of the key)
175 *temp2 = _mm_shuffle_epi32(*temp2, 0xFF); // Broadcast one word from keygen assist
176 temp4 = _mm_slli_si128(*temp1, 0x4);
177 *temp1 = _mm_xor_si128(*temp1, temp4);
178 temp4 = _mm_slli_si128(temp4, 0x4);
179 *temp1 = _mm_xor_si128(*temp1, temp4);
180 temp4 = _mm_slli_si128(temp4, 0x4);
181 *temp1 = _mm_xor_si128(*temp1, temp4);
182 *temp1 = _mm_xor_si128(*temp1, *temp2); // Mix with assist result
183
184 // Expand temp3 (second half of the key)
185 temp4 = _mm_aeskeygenassist_si128(*temp1, 0x00); // Generate assist from updated temp1
186 temp4 = _mm_shuffle_epi32(temp4, 0xAA); // Broadcast relevant word
187
188 *temp3 = _mm_xor_si128(*temp3, _mm_slli_si128(*temp3, 0x4));
189 *temp3 = _mm_xor_si128(*temp3, _mm_slli_si128(*temp3, 0x4));
190 *temp3 = _mm_xor_si128(*temp3, _mm_slli_si128(*temp3, 0x4));
191 *temp3 = _mm_xor_si128(*temp3, temp4); // Final mixing with assist
192}
